1. Node.js + TypeScript

Single-file config. No Dockerfile, no Compose—Features handle everything.

// .devcontainer/devcontainer.json
{
  "name": "Node.js + TypeScript",
  "image": "mcr.microsoft.com/devcontainers/typescript-node:20-bookworm",
  "features": {
    "ghcr.io/devcontainers/features/git:1": {},
    "ghcr.io/devcontainers/features/github-cli:1": {}
  },
  "containerEnv": {
    "NODE_ENV": "development",
    "NODE_OPTIONS": "--max-old-space-size=4096"
  },
  "forwardPorts": [3000, 9229],
  "portsAttributes": {
    "3000": { "label": "App", "onAutoForward": "notify" },
    "9229": { "label": "Debugger", "onAutoForward": "silent" }
  },
  "postCreateCommand": "npm ci && npm run build",
  "postAttachCommand": "npm run dev",
  "customizations": {
    "vscode": {
      "extensions": [
        "dbaeumer.vscode-eslint",
        "esbenp.prettier-vscode",
        "ms-vscode.vscode-typescript-next",
        "bradlc.vscode-tailwindcss",
        "vitest.explorer"
      ],
      "settings": {
        "editor.defaultFormatter": "esbenp.prettier-vscode",
        "editor.formatOnSave": true,
        "editor.codeActionsOnSave": {
          "source.fixAll.eslint": "explicit"
        },
        "typescript.tsdk": "node_modules/typescript/lib",
        "typescript.preferences.importModuleSpecifier": "non-relative"
      }
    }
  }
}
Note postAttachCommand runs every time you reconnect (not just on first create). Perfect for starting the dev server. postCreateCommand runs once after the container is built—use it for npm ci.

2. Python + PostgreSQL (Compose)

// .devcontainer/devcontainer.json
{
  "name": "Python + PostgreSQL",
  "dockerComposeFile": "docker-compose.yml",
  "service": "app",
  "workspaceFolder": "/workspaces/${localWorkspaceFolderBasename}",
  "features": {
    "ghcr.io/devcontainers/features/git:1": {},
    "ghcr.io/devcontainers/features/github-cli:1": {}
  },
  "containerEnv": {
    "DATABASE_URL": "postgresql://dev:dev@db:5432/appdb",
    "PYTHONDONTWRITEBYTECODE": "1",
    "PYTHONUNBUFFERED": "1"
  },
  "forwardPorts": [8000, 5432],
  "portsAttributes": {
    "8000": { "label": "API", "onAutoForward": "notify" },
    "5432": { "label": "PostgreSQL", "onAutoForward": "silent" }
  },
  "postCreateCommand": "pip install -r requirements.txt && python manage.py migrate",
  "postAttachCommand": "python manage.py runserver 0.0.0.0:8000",
  "customizations": {
    "vscode": {
      "extensions": [
        "ms-python.python",
        "ms-python.pylint",
        "ms-python.black-formatter",
        "mtxr.sqltools",
        "mtxr.sqltools-driver-pg"
      ],
      "settings": {
        "python.defaultInterpreterPath": "/usr/local/bin/python",
        "[python]": {
          "editor.defaultFormatter": "ms-python.black-formatter",
          "editor.formatOnSave": true
        },
        "python.testing.pytestEnabled": true,
        "sqltools.connections": [{
          "name": "Dev DB",
          "driver": "PostgreSQL",
          "server": "db",
          "port": 5432,
          "database": "appdb",
          "username": "dev",
          "password": "dev"
        }]
      }
    }
  }
}

# .devcontainer/docker-compose.yml
services:
  app:
    image: mcr.microsoft.com/devcontainers/python:3.12-bookworm
    volumes:
      - ..:/workspaces/${LOCAL_WORKSPACE_FOLDER_BASENAME}:cached
      - pip-cache:/root/.cache/pip
    command: sleep infinity
    depends_on:
      db:
        condition: service_healthy

  db:
    image: postgres:16-alpine
    restart: unless-stopped
    environment:
      POSTGRES_USER: dev
      POSTGRES_PASSWORD: dev
      POSTGRES_DB: appdb
    volumes:
      - pgdata:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U dev -d appdb"]
      interval: 5s
      timeout: 3s
      retries: 5

volumes:
  pgdata:
  pip-cache:
Key detail The depends_on with condition: service_healthy prevents the app from starting before Postgres is ready. Without it, manage.py migrate will fail on first boot. The pip-cache volume survives container rebuilds—subsequent pip install runs are near-instant.

3. Go with Debugging (Delve)

// .devcontainer/devcontainer.json
{
  "name": "Go + Delve",
  "build": {
    "dockerfile": "Dockerfile"
  },
  "features": {
    "ghcr.io/devcontainers/features/git:1": {},
    "ghcr.io/devcontainers/features/docker-in-docker:2": {}
  },
  "containerEnv": {
    "GOPATH": "/go",
    "GOFLAGS": "-race",
    "CGO_ENABLED": "1"
  },
  "forwardPorts": [8080, 2345],
  "portsAttributes": {
    "8080": { "label": "App" },
    "2345": { "label": "Delve", "onAutoForward": "silent" }
  },
  "postCreateCommand": "go mod download && go install github.com/golangci/golangci-lint/cmd/golangci-lint@latest",
  "customizations": {
    "vscode": {
      "extensions": [
        "golang.go",
        "zxh404.vscode-proto3"
      ],
      "settings": {
        "go.lintTool": "golangci-lint",
        "go.lintFlags": ["--fast"],
        "go.testFlags": ["-race", "-count=1"],
        "go.coverOnSave": true,
        "go.coverageDecorator": {
          "type": "gutter"
        }
      }
    }
  }
}

# .devcontainer/Dockerfile
FROM mcr.microsoft.com/devcontainers/go:1.22-bookworm

# Install Delve debugger
RUN go install github.com/go-delve/delve/cmd/dlv@latest

# Install common Go tools
RUN go install golang.org/x/tools/gopls@latest \
 && go install github.com/fatih/gomodifytags@latest \
 && go install github.com/cweill/gotests/gotests@latest

# Pre-download module cache for faster builds
WORKDIR /go/src/app

Setting GOFLAGS="-race" globally enables the race detector for every go test and go run. This catches concurrency bugs early. It makes binaries slower and uses more memory, which is fine for dev—but if you’re running benchmarks, override it with GOFLAGS="" go test -bench=.

Delve setup VS Code’s Go extension auto-detects Delve. To attach to a running process, launch with dlv debug --headless --listen=:2345 --api-version=2 ./cmd/server, then use VS Code’s “Go: Connect to Server” debug config.

4. Rust with cargo watch

// .devcontainer/devcontainer.json
{
  "name": "Rust",
  "build": {
    "dockerfile": "Dockerfile"
  },
  "containerEnv": {
    "CARGO_HOME": "/usr/local/cargo",
    "RUSTUP_HOME": "/usr/local/rustup",
    "SCCACHE_DIR": "/tmp/sccache",
    "RUSTC_WRAPPER": "sccache"
  },
  "mounts": [
    "source=rust-target,target=/workspaces/${localWorkspaceFolderBasename}/target,type=volume",
    "source=rust-cargo-registry,target=/usr/local/cargo/registry,type=volume"
  ],
  "forwardPorts": [8080],
  "postCreateCommand": "cargo fetch",
  "postAttachCommand": "cargo watch -x 'clippy --all-targets' -x 'test' -x 'run'",
  "customizations": {
    "vscode": {
      "extensions": [
        "rust-lang.rust-analyzer",
        "vadimcn.vscode-lldb",
        "tamasfe.even-better-toml",
        "serayuzgur.crates"
      ],
      "settings": {
        "rust-analyzer.check.command": "clippy",
        "rust-analyzer.cargo.buildScripts.enable": true,
        "rust-analyzer.procMacro.enable": true,
        "[rust]": {
          "editor.defaultFormatter": "rust-lang.rust-analyzer",
          "editor.formatOnSave": true
        }
      }
    }
  }
}

# .devcontainer/Dockerfile
FROM mcr.microsoft.com/devcontainers/rust:bookworm

# sccache for compile caching, cargo-watch for auto-rebuild
RUN cargo install sccache cargo-watch cargo-expand

# Common system deps for popular crates (openssl, libpq, etc.)
RUN apt-get update && apt-get install -y \
    pkg-config libssl-dev libpq-dev \
    && rm -rf /var/lib/apt/lists/*

# Pre-warm sccache directory
RUN mkdir -p /tmp/sccache
Foot-gun The target/ directory on a bind mount is painfully slow—tens of thousands of small files. Always use a named volume for target/ (as shown). The trade-off: you lose the build cache when you delete the volume, but day-to-day builds are 3–5x faster. The cargo-registry volume prevents re-downloading crates on every rebuild.

5. Java + Spring Boot + MySQL (Compose)

// .devcontainer/devcontainer.json
{

  "dockerComposeFile": "docker-compose.yml",
  "service": "app",
  "workspaceFolder": "/workspaces/${localWorkspaceFolderBasename}",

  "forwardPorts": [8080, 5005, 3306],
  "portsAttributes": {
    "8080": { "label": "Spring Boot", "onAutoForward": "notify" },
    "5005": { "label": "JDWP Debug", "onAutoForward": "silent" },

# .devcontainer/docker-compose.yml
services:
  app:
    image: mcr.microsoft.com/devcontainers/base:ubuntu
    volumes:
      - ..:/workspaces/${LOCAL_WORKSPACE_FOLDER_BASENAME}:cached
      - gradle-cache:/home/vscode/.gradle
      - m2-cache:/home/vscode/.m2
    command: sleep infinity
    depends_on:
      db:
        condition: service_healthy

  db:
    image: mysql:8.4
    restart: unless-stopped
    environment:
      MYSQL_ROOT_PASSWORD: rootdev
      MYSQL_DATABASE: appdb
      MYSQL_USER: dev
      MYSQL_PASSWORD: dev
    volumes:
      - mysql-data:/var/lib/mysql
    ports:
      - "3306:3306"
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 5s
      timeout: 3s
      retries: 10

volumes:
  mysql-data:
  gradle-cache:
  m2-cache:
Key detail The Gradle and Maven caches are stored in named volumes so a container rebuild doesn’t re-download every dependency. The JAVA_TOOL_OPTIONS sets ZGC with a 2 GB heap—adjust for your Codespaces machine size. For debug, run with ./gradlew bootRun --debug-jvm and attach on port 5005.

The most complex setup. Four services, one Dev Container. You work in the app service (which has both Node and Python); the data stores are sidecar containers.

  app:
    image: mcr.microsoft.com/devcontainers/base:ubuntu
    volumes:
      - ..:/workspaces/${LOCAL_WORKSPACE_FOLDER_BASENAME}:cached
      - node-modules:/workspaces/${LOCAL_WORKSPACE_FOLDER_BASENAME}/web/node_modules

    command: sleep infinity
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy

  db:
    image: postgres:16-alpine
    restart: unless-stopped
    environment:
      POSTGRES_USER: dev
      POSTGRES_PASSWORD: dev
      POSTGRES_DB: appdb
    volumes:
      - pgdata:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U dev -d appdb"]
      interval: 5s
      timeout: 3s
      retries: 5

  redis:
    image: redis:7-alpine
    restart: unless-stopped
    volumes:
      - redis-data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 3s
      retries: 5

volumes:
  pgdata:
  redis-data:
  node-modules:
  pip-cache:

#!/bin/bash
# .devcontainer/setup.sh
set -e

# Backend
echo "==> Installing Python dependencies..."
pip install -r api/requirements.txt

# Frontend
echo "==> Installing Node dependencies..."
cd web && npm ci && cd ..

# Database
echo "==> Running migrations..."
cd api && python manage.py migrate && cd ..

echo "==> Done. Run 'npm run dev' in web/ and 'python manage.py runserver' in api/"
Gotcha With four services, Codespaces needs at least a 4-core machine (8 GB RAM). Set this in .devcontainer/devcontainer.json with "hostRequirements": { "cpus": 4, "memory": "8gb" }. Locally, make sure Docker Desktop has at least 8 GB allocated.
Key detail The node-modules named volume is critical. Without it, node_modules lives on the bind mount—thousands of small files synced between host and container. On macOS, this alone can make next dev take 30+ seconds to start instead of 3.

Lifecycle Hooks Quick Reference

Every config above uses lifecycle hooks. Here is when each one runs:

HookRuns whenUse for
initializeCommand Before container build, on host Login to registries, pull data, host-side checks
onCreateCommand After container create (first time only) Clone submodules, one-time setup
updateContentCommand After create + after source changes (Codespaces prebuild) Rebuild assets, update generated code
postCreateCommand After create (every rebuild) npm ci, pip install, DB migrations
postStartCommand Every container start (including restart) Start background services, print welcome message
postAttachCommand Every time a client (VS Code) attaches Start dev server, tail logs

The most common mistake: putting npm ci in postStartCommand. That runs on every restart, wasting time. Put install commands in postCreateCommand—it runs once per build. Put dev servers in postAttachCommand—they restart when you reconnect.
